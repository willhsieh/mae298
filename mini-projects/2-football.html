## Now, our model

@model function football_matches(home_teams, away_teams, score_home, score_away, teams)

    # Hyper priors
    att ~ Normal(0, 0.1)
    def ~ Normal(0, 0.1)
    att ~ Exponential(1)
    def ~ Exponential(1)
    home ~ Normal(0, 1)
        
    # Team-specific effects
    
    att = zeros(length(teams))
    def = zeros(length(teams))
    
    for i in 1:length(teams)
        att[i] ~ Normal(att, att)
        def[i] ~ Normal(def, def)
    end

    #att ~ filldist(Normal(att, att), length(teams))  # more compact than for-loops
    #def ~ filldist(Normal(def, def), length(teams))
    
    offset = mean(att) + mean(def)
    
    # the number of matches
    n_matches = length(home_teams)

    # scoring rates 
    _home = Vector{Real}(undef, n_matches)     # or just _home = zeros(n_matches)
    _away = Vector{Real}(undef, n_matches)     # or just _away = zeros(n_matches)
        
    # Modeling score-rate and scores for each match
    for i in 1:n_matches
        # scoring rate
        home_team_idx = findfirst(isequal(home_teams[i]), teams)
        away_team_idx = findfirst(isequal(away_teams[i]), teams)

        _home[i] = home + att[home_team_idx] + def[away_team_idx] - offset
        _away[i] = att[away_team_idx] + def[home_team_idx] - offset

        # scores
        score_home[i] ~ Poisson(exp(_home[i]))  # To ensure positive  for Poisson distribution
        score_away[i] ~ Poisson(exp(_away[i]))
    end
end
